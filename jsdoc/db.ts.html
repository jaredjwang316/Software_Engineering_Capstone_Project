* Creates a Postgres client specific to this server request.

    user: PGUSER,
    password: Buffer.from(PGPASSWORD, 'base64').toString('utf-8').trim().replace(/(\r\n|\n|\r)/gm, ""),
    port: parseInt(PGPORT),

})
